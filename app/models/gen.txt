Create User
npx sequelize model:generate --name User --attributes 
name:string,email:string,encryptedPassword:string,role:string,
phone:string,address:string,imageId:string,imageUrl:string,deletedAt:date


Create Ticket
npx sequelize model:generate --name Ticket --attributes
category:string,from:string,to:string,departureTime:date,
returnTime:date,price:INTEGER,flightNumber:string,imageId:string,
imageUrl:string,description:string,createdBy:INTEGER,updatedBy:INTEGER,
deletedAt:date

Create Transaction
npx sequelize model:generate --name Transaction --attributes
userId:INTEGER,ticketId:INTEGER,amount:INTEGER,deletedAt:date

Create Wishlist
npx sequelize model:generate --name Wishlist --attributes
userId:INTEGER,ticketId:INTEGER,deletedAt:date
